文件读写

use std::fs::File;
use std::io::prelude::*;
// 打开文件
let mut file = match File::open(file_path) {
    Ok(f) => f,
    Err(e) => panic!("no such file {} exception:{}", file_path, e),
};
let mut str_val = String::new();
//读取文件
match file.read_to_string(&mut str_val) {
    Ok(s) => s,
    Err(e) => panic!("Error Reading file: {}", e),
};
